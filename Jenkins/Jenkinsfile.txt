pipeline {
    agent any

    environment {
        MLFLOW_TRACKING_USERNAME = 'karrayyessine1'
        MLFLOW_TRACKING_PASSWORD = credentials('dagshub-token')
        MLFLOW_TRACKING_URI = 'https://dagshub.com/karrayyessine1/MLOps_Project.mlflow'
        DAGSHUB_TOKEN = credentials('dagshub-token')
        DAGSHUB_USER = 'karrayyessine1'
        DAGSHUB_REPO = 'MLOps_Project'
    }

    stages {
        stage('Checkout') {
            steps {
                checkout scm
            }
        }

        stage('Setup Environment') {
            steps {
                script {
                    echo '🔧 Creating Virtual Environment...'
                    sh '''
                        python3 -m venv venv
                        . venv/bin/activate
                        pip install --upgrade pip
                        pip install -r backend/src/requirements.txt
                    '''
                }
            }
        }

        stage('Verify Model') {
            steps {
                script {
                    echo '🔍 Verifying model files...'
                    sh '''
                        echo "=== Checking processors directory ==="
                        ls -la notebooks/processors/models/ || echo "❌ Models not found"
                        
                        echo ""
                        echo "=== Checking best model ==="
                        test -f notebooks/processors/models/best_model_final.pkl && echo "✅ Model found" || echo "❌ Model missing"
                    '''
                }
            }
        }

        stage('Test Model') {
            steps {
                script {
                    echo '🧪 Testing model predictions...'
                    sh '''
                        . venv/bin/activate
                        python3 -c "
import pickle
import numpy as np

# Load model
with open('notebooks/processors/models/best_model_final.pkl', 'rb') as f:
    model = pickle.load(f)

# Load test data
with open('notebooks/processors/preprocessed_data.pkl', 'rb') as f:
    data = pickle.load(f)
    X_test = data['X_test'][:5]

# Test prediction
pred = model.predict(X_test)
print('✅ Model prediction test passed')
print(f'Predictions: {pred}')
"
                    '''
                }
            }
        }

        stage('Build Docker Images') {
            steps {
                script {
                    echo '🐳 Building Docker images...'
                    sh '''
                        docker compose build
                    '''
                }
            }
        }

        stage('Deploy') {
            steps {
                script {
                    echo '🚀 Deploying with Docker Compose...'
                    sh '''
                        docker compose down
                        docker compose up -d
                    '''
                }
            }
        }

        stage('Health Check') {
            steps {
                script {
                    echo '🏥 Checking service health...'
                    sh '''
                        sleep 15
                        curl -f http://localhost:8000/health || exit 1
                        echo "✅ Backend is healthy"
                    '''
                }
            }
        }
    }
    
    post {
        always {
            echo '🧹 Cleaning up...'
            sh 'docker compose logs backend || true'
            sh 'docker compose logs frontend || true'
        }
        success {
            echo '✅ Pipeline completed successfully!'
        }
        failure {
            echo '❌ Pipeline failed!'
            sh 'docker compose down || true'
        }
    }
}
